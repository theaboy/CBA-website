    ```typescript
    async function fulfillTicketPurchase(supabase: SupabaseClient, pi: Stripe.PaymentIntent) {
      const { event_id, event_title, quantity, buyer_email, buyer_name } = pi.metadata
      const qty = parseInt(quantity, 10)

      // Idempotency check
      const { data: existing } = await supabase
        .from('orders_ticket')
        .select('id')
        .eq('stripe_payment_intent_id', pi.id)
        .maybeSingle()
      if (existing) return

      // Insert order record
      const { data: order, error: orderError } = await supabase
        .from('orders_ticket')
        .insert({
          event_id,
          stripe_payment_intent_id: pi.id,
          buyer_email,
          buyer_name,
          quantity: qty,
          amount_paid: pi.amount / 100,
        })
        .select('id')
        .single()

      if (orderError) throw orderError

      // Atomically decrement inventory and insert tickets (Postgres RPC)
      const { error: rpcError } = await supabase.rpc('purchase_ticket', {
        p_event_id: event_id,
        p_order_id: order.id,
        p_quantity: qty,
      })

      if (rpcError) {
        if (rpcError.message.startsWith('SOLD_OUT')) {
          // Extremely rare edge case — log and return 200 so Stripe doesn't retry
          console.error('SOLD_OUT after order insert:', rpcError.message)
          return
        }
        throw rpcError
      }

      // Fetch qr_tokens from tickets table (inserted by the RPC)
      const { data: ticketRows } = await supabase
        .from('tickets')
        .select('qr_token')
        .eq('order_ticket_id', order.id)
      const qrTokens = (ticketRows ?? []).map(t => t.qr_token)

      // Query event for email details
      const { data: event } = await supabase
        .from('events')
        .select('date, venue')
        .eq('id', event_id)
        .single()

      await sendTicketPurchaseEmail({
        to: buyer_email,
        buyerName: buyer_name,
        eventTitle: event_title,
        eventDate: event?.date ?? '',
        venue: event?.venue ?? '',
        qrTokens,
      })
    }
    ```
