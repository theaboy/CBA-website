    ```typescript
    export async function sendBeatPurchaseEmail(opts: {
      to: string; buyerName: string; beatTitle: string
      licenseType: string; downloadToken: string
    }) {
      const downloadUrl = `${process.env.NEXT_PUBLIC_SITE_URL}/api/download/${opts.downloadToken}`
      await resend.emails.send({
        from: 'CBA Official <no-reply@cba-official.com>',
        to: opts.to,
        subject: `Your beat is ready — ${opts.beatTitle}`,
        react: BeatPurchaseEmail({ ...opts, downloadUrl }),
      })
    }

    export async function sendTicketPurchaseEmail(opts: {
      to: string; buyerName: string; eventTitle: string
      eventDate: string; venue: string; qrTokens: string[]
    }) {
      await resend.emails.send({
        from: 'CBA Official <no-reply@cba-official.com>',
        to: opts.to,
        subject: `Your ticket — ${opts.eventTitle}`,
        react: TicketPurchaseEmail(opts),
      })
    }

    export async function sendBookingInquiryEmail(opts: {
      adminEmail: string; bookingType: string
      name: string; email: string; details: Record<string, string>
    }) {
      await resend.emails.send({
        from: 'CBA Official <no-reply@cba-official.com>',
        to: opts.adminEmail,
        subject: `New ${opts.bookingType} inquiry — ${opts.name}`,
        react: BookingConfirmationEmail({ name: opts.name, bookingType: opts.bookingType as any }),
      })
    }

    export async function sendBookingConfirmationEmail(opts: {
      to: string; name: string; bookingType: 'studio' | 'dj' | 'beat-inquiry'
    }) {
      await resend.emails.send({
        from: 'CBA Official <no-reply@cba-official.com>',
        to: opts.to,
        subject: `We received your request — CBA Official`,
        react: BookingConfirmationEmail(opts),
      })
    }
    ```
