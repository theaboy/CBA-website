```typescript
import { S3Client, GetObjectCommand, PutObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';

// Use IAM user credentials (NOT IAM role) — role session credentials expire independently
// of expiresIn when deployed on Railway/Render, causing random auth failures.
// Set AWS_ACCESS_KEY_ID and AWS_SECRET_ACCESS_KEY env vars from IAM user (not instance profile).
export const s3Client = new S3Client({
  region: process.env.AWS_REGION ?? 'us-east-1',
  credentials: {
    accessKeyId:     process.env.AWS_ACCESS_KEY_ID ?? '',
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY ?? '',
  },
});

const BUCKET = process.env.S3_BUCKET_NAME ?? 'cba-beats';

/**
 * Generate a time-limited signed GET URL for a preview file.
 * Key must be the S3 object key (e.g. "preview/after-hours-anthem.mp3").
 * Expires in 1 hour (3600 seconds).
 */
export async function getSignedPreviewUrl(key: string): Promise<string> {
  const command = new GetObjectCommand({ Bucket: BUCKET, Key: key });
  return getSignedUrl(s3Client, command, { expiresIn: 3600 });
}

/**
 * Generate a presigned PUT URL for admin browser-direct upload.
 * Expires in 15 minutes (900 seconds).
 * contentType must match the Content-Type header the browser sends on PUT.
 */
export async function getAdminUploadUrl(key: string, contentType: string): Promise<string> {
  const command = new PutObjectCommand({
    Bucket: BUCKET,
    Key: key,
    ContentType: contentType,
  });
  return getSignedUrl(s3Client, command, { expiresIn: 900 });
}
```

No public bucket URLs are ever constructed — only signed URLs from these helpers are used.

**Update src/routes/admin/beats.ts** — add POST /admin/beats/upload-url endpoint.

Add this route BEFORE the existing POST /admin/beats / (create) route, so Express matches /upload-url before /:id wildcards:

```typescript
import { getAdminUploadUrl } from '../../lib/s3';

const uploadUrlSchema = z.object({
  key:         z.string().min(1),         // e.g. "preview/my-beat.mp3" or "full/my-beat.wav"
  contentType: z.string().min(1),         // e.g. "audio/mpeg" or "audio/wav"
});

// POST /admin/beats/upload-url — returns presigned PUT URL for browser-direct S3 upload
adminBeatsRouter.post('/upload-url', async (req, res, next) => {
  try {
    const parsed = uploadUrlSchema.safeParse(req.body);
    if (!parsed.success) {
      throw createError('key and contentType are required', 400);
    }

    // Validate key prefix — only allow preview/ and full/ folders
    const { key, contentType } = parsed.data;
    if (!key.startsWith('preview/') && !key.startsWith('full/') && !key.startsWith('artwork/')) {
      throw createError('key must start with preview/, full/, or artwork/', 400);
    }

    const uploadUrl = await getAdminUploadUrl(key, contentType);
    res.json({ uploadUrl, key, expiresIn: 900 });
  } catch (err) {
    next(err);
  }
});
```

The upload flow is: admin calls POST /admin/beats/upload-url → receives presigned PUT URL → browser PUTs file directly to S3 → admin then calls POST /admin/beats with the S3 key in previewKey/fullKey/artworkKey fields.
